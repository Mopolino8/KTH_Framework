18  SUBROUTINE hbubble
19 C
20 C This routine plots pre-drawn streamline information from file.st3
21 C in a hydrogen bubble like fashion. It is primarily intended for
22 C animation of 3D streamline data.
23 C
24 C The only argument is the three-term vector TCLOCK, which specifies
25 C the start time, end time, and delta time step for the series of
26 C plots. In interactive mode, the user is prompted to advance each
27 C time step. In IFDEMO mode, DMPFRM is called between each time step.
28 C It is assumed that the hydrogen bubble wire burst times have been
29 C specified in TBURST prior to calling this routine.
30 C
31 C pff 31 Jul 90 18:30:23 PDT
32 C
33  include 'basics.inc'
34  include 'basicsp.inc'
35  CHARACTER*1 YESNO
36  LOGICAL IFAUTH,IFMOVI
37 C
38 C Auto mode?
39 C
40  ifauth=.false.
41  CALL prs(' Auto frame advance? (y/n)$')
42  CALL res(yesno,1)
43  IF (yesno.EQ.'Y'.OR.yesno.EQ.'y') ifauth=.true.
44  ifmovi=.false.
45  CALL prs(' Dump frames automatically? (y/n)$')
46  CALL res(yesno,1)
47  IF (yesno.EQ.'Y'.OR.yesno.EQ.'y') ifmovi=.true.
48 C
49 C Check if tclock3 specifies DT or number of clock steps.
50 C
51  IF (tclock(3).GT.1.0) THEN
52  dtc=(tclock(2)-tclock(1))/tclock(3)
53  ELSE
54  dtc=tclock(3)
55  ENDIF
56  IF (tclock(1).EQ.tclock(2).OR.tclock(3).EQ.0.0) THEN
57  nclock=1
58  ELSE
59  nclock=int( 0.5+(tclock(2)-tclock(1))/dtc )+1
60  ENDIF
61 C
62 C Main loop: Draw each BURST as it would be seen at time TCLK0
63 C
64  DO 5000 iclock=1,nclock
65 C Clear and set up the frame - viewpt, zoom, color, etc.
66  CALL setfrm
67  CALL drwire
68 C
69 C==============================================================
70 C Draw the streamlines twice:
71 C==============================================================
72 C
73  tclk0=tclock(1)+dtc*float(iclock-1)
74  DO 2000 iburst=1,nburst
75  tau1=tclk0-tburst(2,iburst)
76  tau2=tclk0-tburst(1,iburst)
77  tau1=max(tau1,0.0)
78  tau2=max(tau2,0.0)
79  IF (tau2.GT.0.0) THEN
80  DO 1000 ip=1,npart
81  CALL drwstl(ip,tau1,tau2,1)
82  1000 CONTINUE
83  ENDIF
84  2000 CONTINUE
85 C
86 C Redraw the object to hide lines behind it
87 C
88  CALL drobjt
89  CALL drwire
90 C
91  DO 4000 iburst=1,nburst
92  tau1=tclk0-tburst(2,iburst)
93  tau2=tclk0-tburst(1,iburst)
94  tau1=max(tau1,0.0)
95  tau2=max(tau2,0.0)
96  IF (tau2.GT.0.0) THEN
97  DO 3000 ip=1,npart
98  CALL drwstl(ip,tau1,tau2,2)
99  3000 CONTINUE
100  ENDIF
101  4000 CONTINUE
102 C
103 C==============================================================
104 C Dump frame?
105 C==============================================================
106 C
107  IF (ifdemo) THEN
108  CALL dmpfrm
109  ELSEIF (.NOT.ifauth) THEN
110  CALL prs(' Advance frame? (y/n)$')
111  CALL res(yesno,1)
112  IF (yesno.EQ.'N'.OR.yesno.EQ.'n') RETURN
113  ENDIF
114  IF (ifmovi.AND..NOT.ifdemo) CALL dmpfrm
115 C==============================================================
116 C Advance to next time step
117 C==============================================================
118  5000 CONTINUE
119  RETURN
120  END

159 C
160  IF (tau1.GE.xstr(4,ipmax)) RETURN

165  frac=0.0
166  ELSE
167  ipglo=ipmin
168  ipgup=ipmax
169  DO 100 j=ipmin,ipmax
170  ipg = (ipglo+ipgup)/2
171  ipg1=ipg+1
172  IF (xstr(4,ipg).LE.tau1.AND.tau1.LT.xstr(4,ipg1)) THEN
173  frac=(tau1-xstr(4,ipg))/(xstr(4,ipg1)-xstr(4,ipg))
174  GOTO 101
175  ENDIF
176 C Else, we didn't find it yet
177  IF (tau1.LT.xstr(4,ipg)) THEN
178  ipgup=ipg
179  ELSE
180  ipglo=ipg1
181  ENDIF
182  100 CONTINUE
183  101 CONTINUE
184  ENDIF
185 C
186 C Start:
187  xp0 = xstr(1,ipg)+frac*(xstr(1,ipg1)-xstr(1,ipg))

190  wp0 = xstr(5,ipg)+frac*(xstr(5,ipg1)-xstr(5,ipg))

193 C Let's draw (while Tau < Tau2)
194 C
195  DO 200 j=ipg1,ipmax
196  IF (xstr(4,j).LE.tau2) THEN
197  xp1 = xstr(1,j)
198  yp1 = xstr(2,j)
199  zp1 = xstr(3,j)
200  wp1 = xstr(5,j)
201  wpp = (wp0+wp1)/2.0
202  icc = icolor1(wpp)
203 c write(6,*) 'trouble:',icc,wpp,wp1,xp1,yp1,zp1,j,ipg1,ipmax
204  CALL color(icc)
205  CALL drawc(xiso(xp1,yp1,zp1),yiso(xp1,yp1,zp1))
206  xp0 = xp1
207  yp0 = yp1
208  zp0 = zp1
209  wp0 = wp1
210  ELSE
211  frac=(tau2-xstr(4,j-1))/(xstr(4,j)-xstr(4,j-1))
212  xpp = xp0+frac*(xstr(1,j)-xp0)
213  ypp = yp0+frac*(xstr(2,j)-yp0)
214  zpp = zp0+frac*(xstr(3,j)-zp0)
215  wpp = wp0+frac*(xstr(5,j)-wp0)
216  icc = icolor1(wpp)
217  CALL color(icc)
218  CALL drawc(xiso(xpp,ypp,zpp),yiso(xpp,ypp,zpp))
219  GOTO 201
220  ENDIF
221  200 CONTINUE
222  201 CONTINUE
223  RETURN
224  END
