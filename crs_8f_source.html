147  subroutine find_match(vlist,iface,ie,jf,je)
148 c
149 c Return list of corresponding vertices when (iface,ie) is
150 c "periodic" with (jf,je).
151 c
152  include 'basics.inc'
153  include 'basicsp.inc'
154 c
155  integer vlist(4,2)
156 c
157  real xl(4,2)
158  real yl(4,2)
159  real zl(4,2)
160 c
161 c order3 is an array which gives a cyclic permutation of vertices in
162 c counter-clockwise order when looking at a face from *outside* the cube.
163 c Using pre-nek ordering, which is miserably non-symmetric...
164  integer order3(4,6)
165  save order3
166  data order3 / 1, 2, 6, 5
167  $ , 2, 3, 7, 6
168  $ , 3, 4, 8, 7
169  $ , 4, 1, 5, 8
170  $ , 1, 4, 3, 2
171  $ , 5, 6, 7, 8 /
172 c
173 c Pairing of face (jf,je) to (iface,ie) is done by considering
174 c 4 rotations of (jf,je) and taking that which minimizes the
175 c L2 measure of the separation of the vertices.
176 c
177  nc = 2**ndim
178  ncf = 2**(ndim-1)
179 c
180 c Load (iface,ie) in reverse order
181 c
182  do ic = 1,ncf
183  xl(ic,1) = x(order3(ncf+1-ic,iface),ie)
184  yl(ic,1) = y(order3(ncf+1-ic,iface),ie)
185  zl(ic,1) = z(order3(ncf+1-ic,iface),ie)
186  vlist(ic,1) = order3(ncf+1-ic,iface) + nc*(ie-1)
187  enddo
188 c
189  dmin = 1.e22
190  jrmn = 0
191  do jrot = 0,ncf-1
192  do jc=1,ncf
193  jcr = mod1(jc+jrot,ncf)
194  xl(jc,2) = x(order3(jcr,jf),je)
195  yl(jc,2) = y(order3(jcr,jf),je)
196  zl(jc,2) = z(order3(jcr,jf),je)
197  enddo
198  if (ndim.eq.2) call rzero(zl,8)
199  d = 0.
200  do ic=1,ncf !major error found 11/19/01 pff
201  d = d + (xl(ic,1)-xl(ic,2))**2
202  $ + (yl(ic,1)-yl(ic,2))**2
203  $ + (zl(ic,1)-zl(ic,2))**2
204  enddo
205  if (d.lt.dmin) then
206  dmin = d
207  jrmn = jrot
208  endif
209  enddo
210 c
211 c OK... distance minimizing rotation now found. Choose this
212 c as list
213 c
214  do jc = 1,ncf
215  jcr = mod1(jc+jrmn,ncf)
216  vlist(jc,2) = order3(jcr,jf) + nc*(je-1)
217  enddo
218 c write(6,1) jrmn,'rot1',(vlist(k,1),k=1,4),ie,iface
219 c write(6,2) 'rot2',(vlist(k,2),k=1,4),je,jf,dmin
220 c 1 format(i1,a4,2x,6i7)
221 c 2 format(1x,a4,2x,6i7,1pe12.4)
222 c
223  return
224  end
