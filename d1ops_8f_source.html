11 c jp = interpolation matrix, mapping from pressure nodes to velocity
12 c z = GLL points
13 c n = polynomial degree (velocity space)

15 c

17 c
18 c

56  subroutine fd_weights_full(xx,x,n,m,c)
57 c
58 c This routine evaluates the derivative based on all points
59 c in the stencils. It is more memory efficient than "fd_weights"
60 c
61 c This set of routines comes from the appendix of
62 c A Practical Guide to Pseudospectral Methods, B. Fornberg
63 c Cambridge Univ. Press, 1996. (pff)
64 c
65 c Input parameters:
66 c xx -- point at wich the approximations are to be accurate
67 c x -- array of x-ordinates: x(0:n)
68 c n -- polynomial degree of interpolant (# of points := n+1)
69 c m -- highest order of derivative to be approxxmated at xi
70 c
71 c Output:
72 c c -- set of coefficients c(0:n,0:m).
73 c c(j,k) is to be applied at x(j) when
74 c the kth derivative is approxxmated by a
75 c stencil extending over x(0),x(1),...x(n).
76 c
77 c
78  real x(0:n),c(0:n,0:m)
79 c
80  c1 = 1.
81  c4 = x(0) - xx
82 c
83  do k=0,m
84  do j=0,n
85  c(j,k) = 0.
86  enddo
87  enddo
88  c(0,0) = 1.
89 c
90  do i=1,n
91  mn = min(i,m)
92  c2 = 1.
93  c5 = c4
94  c4 = x(i)-xx
95  do j=0,i-1
96  c3 = x(i)-x(j)
97  c2 = c2*c3
98  do k=mn,1,-1
99  c(i,k) = c1*(k*c(i-1,k-1)-c5*c(i-1,k))/c2
100  enddo
101  c(i,0) = -c1*c5*c(i-1,0)/c2
102  do k=mn,1,-1
103  c(j,k) = (c4*c(j,k)-k*c(j,k-1))/c3
104  enddo
105  c(j,0) = c4*c(j,0)/c3
106  enddo
107  c1 = c2
108  enddo
109 c call outmat(c,n+1,m+1,'fdw',n)
110  return
111  end
