222  subroutine convert
223 c
224 c Subroutine to convert an already read exodusII mesh to a nek
225 c mesh. The idea is to fill each element's node coordinates of
226 c size lx1**3 (3D) or lx1**2 (2D) (lx1=3) with the hex27/quad9
227 c coordinates.
228 c
229  include 'exodusII.inc'
230  include 'SIZE'
231 c
232 c node and face conversion (it works at least for cubit):
233 c
234  integer exo_to_nek_vert3D(27)
235  data exo_to_nek_vert3d
236  & / 19, 1, 7, 25, 21, 3, 9, 27, 10 ! hex27 to nek numbering
237  & , 4, 16, 22, 20, 2, 8, 26, 12, 6
238  & , 18, 24, 14, 13, 15, 23, 5, 11, 17 /
239 
240  integer exo_to_nek_vert2D(9)
241  data exo_to_nek_vert2d / 1, 3, 9, 7, 2, 6, 8, 4, 5 / ! quad9 to nek numbering
242 
243  integer exo_to_nek_face3D(6)
244  data exo_to_nek_face3d / 1, 5, 3, 6, 4, 2 / ! symmetric face numbering
245 
246  integer exo_to_nek_face2D(4)
247  data exo_to_nek_face2d / 1, 2, 3, 4 / ! symmetric face numbering
248 
249  write(6,*)
250  write(6,'(A)') 'Converting elements ...'
251  write(6,*)
252 
253  nvert = 3**num_dim
254 
255  do iel = 1, num_elem
256  if (mod(iel,100).eq.0.or.iel.eq.num_elem) then
257  write(6,'(A,I8)') 'Reading element ',iel
258  endif
259  do ivert = 1, nvert
260  if (num_dim.eq.2) then
261  jvert = exo_to_nek_vert2d(ivert)
262  xm1(jvert,1,1,iel)=x_exo(connect(nvert*(iel-1)+ivert))
263  ym1(jvert,1,1,iel)=y_exo(connect(nvert*(iel-1)+ivert))
264  else
265  jvert = exo_to_nek_vert3d(ivert)
266  xm1(jvert,1,1,iel)=x_exo(connect(nvert*(iel-1)+ivert))
267  ym1(jvert,1,1,iel)=y_exo(connect(nvert*(iel-1)+ivert))
268  zm1(jvert,1,1,iel)=z_exo(connect(nvert*(iel-1)+ivert))
269  endif
270  enddo
271  enddo
272 c zero-out bc and curve sides arrays
273  call blank (cbc,3*2*ldim*max_num_elem)
274  call rzero (bc,5*2*ldim*max_num_elem)
275  call blank (ccurve,(4+8*(ldim-2))*max_num_elem)
276  call rzero (curve,2*ldim*12*max_num_elem)
277 c
278 c set bc's
279 c
280  if (num_side_sets.eq.0) return ! no sidesets
281 
282  write(6,'(a)') ""
283  write(6,'(a)') "Converting SidSets ..."
284 c the expensive part, improve it...
285  do iss=1,num_side_sets ! loop over ss
286  write(6,'(a)') ""
287  write(6,'(a,i2)') "Sideset ",idss(iss)
288  do iel=1,num_elem
289  if (mod(iel,100).eq.0.or.iel.eq.num_elem) then
290  write(6,'(A,I8)') ' Element ',iel
291  endif
292  do ifc=1,2*num_dim ! loop over faces
293  do i=1,num_sides_in_set(iss) ! loop over sides in ss
294  if ( (iel.eq.elem_list(i,iss))
295  & .and. (ifc.eq.side_list(i,iss)) ) then
296  if (num_dim.eq.2) then
297  jfc = exo_to_nek_face2d(ifc)
298  else
299  jfc = exo_to_nek_face3d(ifc)
300  endif
301  cbc(jfc,iel) = 'EXO' ! dummy exodus bc
302  bc(5,jfc,iel) = idss(iss)
303  endif
304  enddo
305  enddo
306  enddo
307  enddo
308 
309  return
310  end

512  subroutine gen_rea_midside_e(e)
513 
514  include 'SIZE'
515 
516  real len
517  real x3(27),y3(27),z3(27),xyz(3,3)
518  character*1 ccrve(12)
519  integer e,edge
520 
521  integer e3(3,12)
522  save e3
523  data e3 / 1, 2, 3, 3, 6, 9, 9, 8, 7, 7, 4, 1
524  $ , 19,20,21, 21,24,27, 27,26,25, 25,22,19
525  $ , 1,10,19, 3,12,21, 9,18,27, 7,16,25 /
526 
527  call chcopy(ccrve,ccurve(1,e),12)
528 
529  call map2reg (x3,3,xm1(1,1,1,e),1) ! Map to 3x3x3 array
530  call map2reg (y3,3,ym1(1,1,1,e),1)
531  if (num_dim.eq.3) call map2reg (z3,3,zm1(1,1,1,e),1)
532 
533 c Take care of spherical curved face defn
534  if (ccurve(5,e).eq.'s') then
535  call chcopy (ccrve(1),'ssss',4) ! face 5
536  call chcopy (ccrve(5),' ',1) ! face 5
537  endif
538  if (ccurve(6,e).eq.'s') then
539  call chcopy (ccrve(5),'ssss',4) ! face 6
540  endif
541 
542  tol = 1.e-4
543  tol2 = tol**2
544  nedge = 4 + 8*(num_dim-2)
545 
546  do i=1,nedge
547  if (ccrve(i).eq.' ') then
548  do j=1,3
549  xyz(1,j) = x3(e3(j,i))
550  xyz(2,j) = y3(e3(j,i))
551  xyz(3,j) = z3(e3(j,i))
552  enddo
553  len = 0.
554  h = 0.
555  do j=1,num_dim
556  xmid = .5*(xyz(j,1)+xyz(j,3))
557  h = h + (xyz(j,2)-xmid)**2
558  len = len + (xyz(j,3)-xyz(j,1))**2
559  enddo
560  if (h.gt.tol2*len) ccurve(i,e) = 'm'
561  if (h.gt.tol2*len) call copy (curve(1,i,e),xyz(1,2),num_dim)
562  endif
563  enddo
564 
565  return
566  end

717  subroutine fd_weights_full(xx,x,n,m,c)
718 c
719 c This routine evaluates the derivative based on all points
720 c in the stencils. It is more memory efficient than "fd_weights"
721 c
722 c This set of routines comes from the appendix of
723 c A Practical Guide to Pseudospectral Methods, B. Fornberg
724 c Cambridge Univ. Press, 1996. (pff)
725 c
726 c Input parameters:
727 c xx -- point at wich the approximations are to be accurate
728 c x -- array of x-ordinates: x(0:n)
729 c n -- polynomial degree of interpolant (# of points := n+1)
730 c m -- highest order of derivative to be approxxmated at xi
731 c
732 c Output:
733 c c -- set of coefficients c(0:n,0:m).
734 c c(j,k) is to be applied at x(j) when
735 c the kth derivative is approxxmated by a
736 c stencil extending over x(0),x(1),...x(n).
737 c
738 c
739  real x(0:n),c(0:n,0:m)
740 c
741  c1 = 1.
742  c4 = x(0) - xx
743 c
744  do k=0,m
745  do j=0,n
746  c(j,k) = 0.
747  enddo
748  enddo
749  c(0,0) = 1.
750 c
751  do i=1,n
752  mn = min(i,m)
753  c2 = 1.
754  c5 = c4
755  c4 = x(i)-xx
756  do j=0,i-1
757  c3 = x(i)-x(j)
758  c2 = c2*c3
759  do k=mn,1,-1
760  c(i,k) = c1*(k*c(i-1,k-1)-c5*c(i-1,k))/c2
761  enddo
762  c(i,0) = -c1*c5*c(i-1,0)/c2
763  do k=mn,1,-1
764  c(j,k) = (c4*c(j,k)-k*c(j,k-1))/c3
765  enddo
766  c(j,0) = c4*c(j,0)/c3
767  enddo
768  c1 = c2
769  enddo
770 
771  return
772  end
