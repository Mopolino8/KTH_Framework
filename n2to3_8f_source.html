1301  subroutine sweep_circ(xc,yc,zc,x,y,n,theta) ! z0=theta
1302 
1303  real xc(n),yc(n),zc(n),x(n),y(n)
1304 
1305 c project x() and y() into theta plane parallel to z-axis
1306 
1307 c Intitial convention: x=r, y=z
1308 
1309  one = 1.

1311  th = -pi*theta/180. ! Flip sweep direction to preserve right-handedness
1312

1316  yc(i) = rad*sin(th)
1317  zc(i) = y(i)
1318  enddo
