211 !******************************************************************************
212 !
213 ! Purpose: set outflow boundary condition for one cell.
214 !
215 ! Description: the subsonic boundary condition is based on non-reflecting,
216 ! characteristics method of Whitfield and Janus: Three-Dimensional
217 ! Unsteady Euler Equations Solution Using Flux Vector Splitting.
218 ! AIAA Paper 84-1552, 1984. The supersonic boundary condition
219 ! consists of simple extrapolation.
220 !
221 ! Input: bcOpt = boundary treatment: subsonic, supersonic, or mixed
222 ! pout = given static outlet pressure
223 ! sx/y/zn = components of ortho-normalized face vector (outward facing)
224 ! cpgas = specific heat at constant pressure (boundary cell)
225 ! mol = molecular mass at boundary cell
226 ! rho = density at boundary cell
227 ! rhou/v/w = density * velocity components at boundary cell
228 ! rhoe = density * total energy at boundary cell
229 ! press = static pressure at boundary cell
230 !
231 ! Output: rhob = density at boundary
232 ! rhou/v/wb = density * velocity components at boundary
233 ! rhoeb = density * total energy at boundary
234 !
235 ! Notes: this condition is valid only for thermally and calorically
236 ! perfect gas (supersonic outflow valid for all gases).
237 !
238 !******************************************************************************
239 !
240 ! $Id: BcondOutflowPerf.F90,v 1.1.1.1 2014/05/05 21:47:47 tmish Exp $
241 !
242 ! Copyright: (c) 2002 by the University of Illinois
243 !

246  SUBROUTINE bcondoutflowperf( bcOpt,pout,sxn,syn,szn,cpgas,mol,
247  > rho,rhou,rhov,rhow,rhoe,press,
248  > rhob,rhoub,rhovb,rhowb,rhoeb )
249 
250  IMPLICIT NONE
251  integer bcopt_subsonic,bcopt_mixed
252  parameter(bcopt_subsonic=1)
253  parameter(bcopt_mixed=0)
254  real MixtPerf_C_DGP,MixtPerf_Eo_DGPUVW,MixtPerf_G_CpR,MixtPerf_R_M
255  > , mixtperf_p_deogvm2
256  external mixtperf_c_dgp,mixtperf_eo_dgpuvw,mixtperf_g_cpr,
257  > mixtperf_r_m,mixtperf_p_deogvm2
258 
259 ! ... parameters
260  INTEGER bcOpt
261 
262  REAL pout
263  REAL rho, rhou, rhov, rhow, rhoe, press
264  REAL sxn, syn, szn, cpgas, mol
265  REAL rhob, rhoub, rhovb, rhowb, rhoeb
266 
267 ! ... local variables
268  REAL csound, rgas, gamma, gam1, u, v, w, mach, rrhoc, deltp,
269  > ub, vb, wb, vnd
270 
271 !******************************************************************************
272 ! gas properties; velocity components; Mach number
273 
274  rgas = mixtperf_r_m( mol )
275  gamma = mixtperf_g_cpr( cpgas,rgas )
276  gam1 = gamma - 1.0
277 
278  u = rhou/rho
279  v = rhov/rho
280  w = rhow/rho
281  csound = mixtperf_c_dgp( rho,gamma,press )
282  mach = sqrt(u*u+v*v+w*w)/csound
283 
284 ! subsonic flow ---------------------------------------------------------------
285 
286  IF (mach .lt. 1.0 .AND.
287  >(bcopt .eq. bcopt_subsonic .OR. bcopt .eq. bcopt_mixed)) THEN
288  rrhoc = 1.0/(rho*csound)
289  deltp = press - pout
290  rhob = rho - deltp/(csound*csound)
291  ub = u + sxn*deltp*rrhoc
292  vb = v + syn*deltp*rrhoc
293  wb = w + szn*deltp*rrhoc
294 
295 ! - special treatment to prevent "deltp" from changing the sign
296 ! of velocity components. This may happen for very small u, v, w.
297 
298  vnd = ub*sxn + vb*syn + wb*szn
299  IF ( vnd .lt. 0.0 ) THEN ! inflow at outflow boundary
300  ub = sign(1.0,u)*max(abs(ub),abs(u))
301  vb = sign(1.0,v)*max(abs(vb),abs(v))
302  wb = sign(1.0,w)*max(abs(wb),abs(w))
303  END IF ! vnd
304 
305  rhoub = rhob*ub
306  rhovb = rhob*vb
307  rhowb = rhob*wb
308  rhoeb = rhob*mixtperf_eo_dgpuvw( rhob,gamma,pout,ub,vb,wb )
309 
310 ! supersonic flow -------------------------------------------------------------
311 
312  ELSE
313  rhob = rho
314  rhoub = rhou
315  rhovb = rhov
316  rhowb = rhow
317  rhoeb = rhoe
318  END IF ! mach
319 
320  end
