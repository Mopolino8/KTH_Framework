2141  subroutine sgefa(a,lda,n,ipvt,info)
2142  integer lda,n,ipvt(1),info

2144 c
2145 c sgefa factors a real matrix by gaussian elimination.
2146 c
2147 c sgefa is usually called by sgeco, but it can be called
2148 c directly with a saving in time if rcond is not needed.
2149 c (time for sgeco) = (1 + 9/n)*(time for sgefa) .
2150 c
2151 c on entry
2152 c
2153 c a real(lda, n)
2154 c the matrix to be factored.
2155 c
2156 c lda integer
2157 c the leading dimension of the array a .
2158 c
2159 c n integer
2160 c the order of the matrix a .
2161 c
2162 c on return
2163 c
2164 c a an upper triangular matrix and the multipliers
2165 c which were used to obtain it.
2166 c the factorization can be written a = l*u where
2167 c l is a product of permutation and unit lower
2168 c triangular matrices and u is upper triangular.
2169 c
2170 c ipvt integer(n)
2171 c an integer vector of pivot indices.
2172 c
2173 c info integer
2174 c = 0 normal value.
2175 c = k if u(k,k) .eq. 0.0 . this is not an error
2176 c condition for this subroutine, but it does
2177 c indicate that sgesl or sgedi will divide by zero
2178 c if called. use rcond in sgeco for a reliable
2179 c indication of singularity.
2180 c
2181 c linpack. this version dated 08/14/78 .
2182 c cleve moler, university of new mexico, argonne national lab.
2183 c
2184 c subroutines and functions
2185 c
2186 c blas saxpy,sscal,isamax
2187 c
2188 c internal variables
2189 c
2190  real t
2191  integer isamax,j,k,kp1,l,nm1
2192 c
2193 c
2194 c gaussian elimination with partial pivoting
2195 c
2196  info = 0
2197  nm1 = n - 1
2198  if (nm1 .lt. 1) go to 70
2199  do 60 k = 1, nm1
2200  kp1 = k + 1
2201 c
2202 c find l = pivot index
2203 c
2204  l = isamax(n-k+1,a(k,k),1) + k - 1
2205  ipvt(k) = l
2206 c
2207 c zero pivot implies this column already triangularized
2208 c
2209  if (a(l,k) .eq. 0.0e0) go to 40
2210 c
2211 c interchange if necessary
2212 c
2213  if (l .eq. k) go to 10
2214  t = a(l,k)
2215  a(l,k) = a(k,k)
2216  a(k,k) = t
2217  10 continue
2218 c
2219 c compute multipliers
2220 c
2221  t = -1.0e0/a(k,k)
2222  call sscal(n-k,t,a(k+1,k),1)
2223 c
2224 c row elimination with column indexing
2225 c
2226  do 30 j = kp1, n
2227  t = a(l,j)
2228  if (l .eq. k) go to 20
2229  a(l,j) = a(k,j)
