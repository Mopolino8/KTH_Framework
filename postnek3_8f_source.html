1241 c
1242  return
1243  endif
1244 c
1245 c
1246 c
1247 c
1248 c Else ... 3D case....
1249 c
1250 c a d e
1251 c Get symmetric 3x3 matrix d b f
1252 c e f c
1253 c
1254  a = aa(1,1)
1255  b = aa(2,2)
1256  c = aa(3,3)
1257  d = 0.5*(aa(1,2)+aa(2,1))
1258  e = 0.5*(aa(1,3)+aa(3,1))
1259  f = 0.5*(aa(2,3)+aa(3,2))
1260  ef = e*f
1261  df = d*f
1262  f2 = f*f
1263 c
1264 c Use Gershgorin to get bounds on eigenvalues
1265 c
1266  x0 = a - abs(d) - abs(e)
1267  x0 = min(x0,(b - abs(d) - abs(f)))
1268  x0 = min(x0,(c - abs(e) - abs(f)))
1269  x1 = a + abs(d) + abs(e)
1270  x1 = max(x1,(b + abs(d) + abs(f)))
1271  x1 = max(x1,(c + abs(e) + abs(f)))
1272 c
1273  dx = x1-x0
1274  x2 = 0.5*(x0+x1)
1275  tol = 1.e-4*(abs(x0)+abs(x1)+abs(x2))
1276 c
1277  if (dx.le.tol) then
1278  lam2 = x2
1279  return
1280  endif
1281 c
1282 c Increase interval width to guarantee bracketing
1283  x0 = x0-dx*1.e-3
1284  x1 = x1+dx*1.e-3
1285 c
1286 c number of known roots is zero
1287  nr = 0
1288 c
1289  tolf = 1.e-5
1290  miter = 0
1291 c
1292  do i=1,3
1293  if (i.eq.1) then
1294 c call muller(lam(i),x2,x1,x0,tol,tolf,kiter,ierr)
1295  lam(i) = zbrent(x1,x0,tol,tolf,kiter,ierr)
1296  else
1297  call muller2(lam(i),x2,x1,x0,tol,tolf,kiter,i,ierr)
1298  endif
1299  if (ierr.ne.0) return
1300  miter = max(kiter,miter)
1301  lam2=lam(i)
1302  if (i.eq.2 .and. lam(1).gt.lam(2)) then
1303  lam(2)=lam(1)
1304  lam(1)=lam2
1305  endif
1306 c
1307 c Shift initial guesses away from previous solution
1308 c
1309  if (abs(x0-lam2).lt.tol) then
1310  x0 = x0 - 0.1*(x2-x0)
1311  elseif (abs(x1-lam2).lt.tol) then
1312  x1 = x1 + 0.1*(x2-x0)
1313  elseif (abs(x2-lam2).lt.tol) then
1314  x2 = x2 + 0.5*(x1-x2)
1315  endif
1316 c
1317  r0 = lam(1)
1318  r1 = lam(2)
1319  nr = i
1320  enddo
1321 c
1322 c write(6,1) (aa(k,1),k=1,9),(lam(l),l=1,3)
1323 c 1 format(1p12e10.2)

1327 c
1328 c

1331  return

1334  return

1341  subroutine muller(x2,x2i,x1i,x0i,tol,tolf,kiter,ierr)
1342 c
1343 c Use muller's method to find (known real) roots of f(x)
1344 c Burden and Faires p. 79
1345 c
1346  common /ecmnl/ iffout,ifdefl
1347  logical iffout,ifdefl
1348 c
1349  iffout = .false.
1350  ifdefl = .true.
1351  ierr = 0

1354  x1=x1i

1356  kiter=1
1357 c
1358  1 continue

1360  h2=x2-x1

1363  if (abs(f0).lt.tolf) then

1366  endif
1367 c

1369  if (abs(f1).lt.tolf) then
1370  x2 = x1
1371  return

1373 c
1374  f2 = ff(x2)
1375  if (abs(f2).lt.tolf) return

1378  d = (d2-d1)/(h1+h2)
1379 c

1381 c
1382  b = d2+h2*d
1383  disc = b*b-4.*f2*d

1385  x2 = x2+0.5*min(abs(x1-x2),abs(x0-x2))

1387  endif

1389  do i=1,9
1390  disc = sqrt(disc)
1391  bm = b-disc
1392  bp = b+disc
1393  if (abs(bm).lt.abs(bp)) then
1394  e=bp
1395  else
1396  e=bm
1397  endif
1398  h = -2*f2/e
1399  x0 = x1
1400  x1 = x2
1401  x2 = x2+h
1402 c
1403  f0 = f1
1404  f1 = f2
1405 c
1406  3 continue
1407 c
1408  if (abs(h).lt.tol) then
1409  return
1410  endif
1411  kiter = kiter+1
1412  if (kiter.gt.9) ifdefl=.false.
1413 c
1414  h1=x1-x0
1415  h2=x2-x1
1416  f2 = ff(x2)
1417  if (abs(f2).lt.tolf) return
1418  d1 = (f1-f0)/h1
1419  d2 = (f2-f1)/h2
1420  d = (d2-d1)/(h1+h2)
1421 c
1422 c Check discriminate
1423 c
1424  b = d2+h2*d
1425  disc = b*b-4.*f2*d
1426  if (abs(disc).lt.tol) disc = tol*tol
1427  if (disc.eq.0.) disc = 1.e-15
1428  if (disc.lt.0) then
1429  write(6,4) i,kiter,disc,x0,x1,x2,f0,f1,f2
1430  4 format(2i2,1p7e10.2)
1431  x2 = x2+1.0*max(abs(x1-x2),abs(x0-x2))
1432  iffout = .true.
1433  if (kiter.gt.15) ierr = 1
1434  if (kiter.gt.15) return
1435  goto 3
1436  endif
1437  enddo
1438  return
1439  end

2028  FUNCTION zbrent(X1,X2,TOL,TOLF,iter)
2029 C
2030 C Using the Van Wijngaarden-Dekker-Brent Method, find the root
2031 C of a function FUNC known to lie between X1 and X2. The root,
2032 C returned as ZBRENT, will be refined until its accuracy is TOL.
2033 C
2034  parameter(itmax=100,eps=3.0e-8)
2035  common /ecmnl/ iffout,ifdefl
2036  logical iffout,ifdefl
2037 C
2038  iffout = .false.
2039  ifdefl = .true.
2040 c
2041  iter=0
2042  a = x1
2043  b = x2
2044  fa = ff(a)
2045  fb = ff(b)
2046 c
2047  if (abs(fa).le.tolf) then
2048  zbrent = a
2049  return
2050  endif
2051  if (abs(fb).le.tolf) then
2052  zbrent = b
2053  return
2054  endif
2055 c
2056  IF (fb*fa.GT.0.0) GOTO 9000
2057  fc=fb
2058 C
2059  DO 1000 iter=1,itmax
2060  IF (fb*fc.GT.0.0) THEN
2061  c = a
2062  fc = fa
2063  d = b-a
2064  e = d
2065  ENDIF
2066  IF (abs(fc).LT.abs(fb)) THEN
2067  a = b
2068  b = c
2069  c = a
2070  fa = fb
2071  fb = fc
2072  fc = fa
2073  ENDIF
2074  tol1 = 2.0*eps*abs(b)+0.5*tol
2075  xm = 0.5*(c-b)
2076  IF (abs(xm).LE.tol1.OR.fb.EQ.0.0) THEN
2077  zbrent = b
2078  return
2079  ENDIF
2080  IF (abs(e).GT.tol1.AND. abs(fa).GT.abs(fb)) THEN
2081 C
2082 C Attempt inverse quadratic interpolation
2083 C
2084  s=fb/fa
2085  IF (a.EQ.c) THEN
2086  p=2.0*xm*s
2087  q=1.0-s
2088  ELSE
2089  q=fa/fc
2090  r=fb/fc
2091  p=s*( 2.0*xm*q*(q-r) - (b-a)*(r-1.0) )
2092  q=(q-1.0)*(r-1.0)*(s-1.0)
2093  ENDIF
2094 C
2095 C Check whether in bounds...
2096 C
2097  IF (p.GT.0.0) q = -q
2098  p = abs(p)

2100 C Accept interpolation.
2101  e=d
2102  d=p/q
2103  ELSE
2104 C Interpolation failed, us bisection.
2105  d=xm
2106  e=d
2107  ENDIF
2108  ELSE
2109 C Bounds decreasing too slowly, use bisection.
2110  d=xm
2111  e=d
2112  ENDIF
2113  a=b
2114  fa=fb
2115  IF (abs(d).GT.tol1) THEN
2116 C Evaluate new trial root
2117  b=b+d
2118  ELSE
2119  b=b+sign(tol1,xm)
2120  ENDIF
2121  fb=ff(b)
2122  1000 CONTINUE
2123 C
2124 C
2125  9000 CONTINUE
2126 c
2127  zbrent=b
2128  if (abs(fa).lt.abs(fb)) zbrent = a
2129 c
2130 c iffout = .true.
2131 c FA = ff(A)
2132 c FB = ff(B)
2133 c write(6,6) 'zbrent',iter,a,b,fa,fb
2134 c if (iffout) write(6,6) 'ff: ',f1,f2,f3,x,r0,r1
2135  6 format(a6,i5,1p6e12.4)
2136 c

2707  subroutine build_new_filter(filt,zpts,nx,kut,wght,nid)
2708 c
2709 c This routing builds a 1D filter with a transfer function that
2710 c looks like:
2711 c
2712 c
2713 c ^
2714 c d_k |
2715 c | |
2716 c 1 |__________ _v_
2717 c | -_
2718 c | \ wght
2719 c | \ ___
2720 c | | ^
2721 c 0 |-------------|---|>
2722 c
2723 c 0 c N k-->
2724 c
2725 c Where c := N-kut is the point below which d_k = 1.
2726 c
2727 c
2728 c
2729 c Here, nx = number of points
2730 c
2731  real filt(nx,nx),zpts(nx)
2732 c
2733  parameter(lm=90)
2734  parameter(lm2=lm*lm)
2735  real phi(lm2),pht(lm2),diag(lm2),rmult(lm),Lj(lm)
2736  integer indr(lm),indc(lm),ipiv(lm)
2737 c
2738  if (nx.gt.lm) then
2739  write(6,*) 'ABORT in build_new_filter:',nx,lm
2740  call exitt
2741  endif
2742 c
2743  kj = 0
2744  n = nx-1
2745  do j=1,nx
2746  z = zpts(j)
2747  call legendre_poly(lj,z,n)
2748  kj = kj+1
2749  pht(kj) = lj(1)
2750  kj = kj+1
2751  pht(kj) = lj(2)
2752  do k=3,nx
2753  kj = kj+1
2754  pht(kj) = lj(k)-lj(k-2)
2755  enddo
2756  enddo
2757  call transpose (phi,nx,pht,nx)
2758  call copy (pht,phi,nx*nx)
2759  call gaujordf (pht,nx,nx,indr,indc,ipiv,ierr,rmult)
2760 c
2761 c Set up transfer function
2762 c
2763  call ident (diag,nx)
2764 c
2765  k0 = nx-kut
2766  do k=k0+1,nx
2767  kk = k+nx*(k-1)
2768  amp = wght*(k-k0)*(k-k0)/(kut*kut) ! quadratic growth
2769  diag(kk) = 1.-amp
2770  enddo
2771 c
2772  call mxm (diag,nx,pht,nx,filt,nx) ! -1
2773  call mxm (phi ,nx,filt,nx,pht,nx) ! V D V
2774  call copy (filt,pht,nx*nx)
2775 c
2776  do k=1,nx*nx
2777  pht(k) = 1.-diag(k)
2778  enddo
2779  np1 = nx+1
2780  if (nid.eq.0) then
2781  write(6,6) 'filt amp',(pht(k),k=1,nx*nx,np1)
2782  write(6,6) 'filt trn',(diag(k),k=1,nx*nx,np1)
2783  6 format(a8,16f7.4,6(/,8x,16f7.4))
2784  endif
2785 c
2786  return
2787  end

2791 c
2792  do i=1,n*n
2793  a(i,1) = 0.
2794  enddo
2795 c
2796  do i=1,n
2797  a(i,i) = 1.
2798  enddo
2799  return
2800  end
