505  subroutine advcell
506  $ (rc,sc,ifedgb,ifcrn,iedgc,val,wrk,ncol,icel,jcel,jcel0)
507 C
508 C Given a cell number, denoted by (ICEL,JCEL), and an inbound
509 C edge (IEDGC=1,2,3,4) where contour crosses into CELL, this
510 C routine finds the exiting point, denoted by (RC,SC). In addition,
511 C (IEDGC,ICEL,JCEL) is updated to represent the new edge.
512 C
513 C If the associated "move" (ie, the line connecting (RC,SC) with the
514 C returned (RC',SC') ), is along the edge of the BOX of CELLS, then
515 C IFEDGB is returned as true.
516 C
517 C If the associated move is an edge move which includes the corner
518 C or vertex of the new cell, IFCRN is returned as true. This is
519 C useful in discriminating edge cell sides which require 2 moves, eg,
520 C as in the cell denoted by "o" below:
521 C /
522 C + + +/ +
523 C /
524 C o /
525 C ----+----+--/ + +
526 C
527  dimension wrk(ncol,ncol)
528  LOGICAL IFEDGB,IFCRN
529  include 'basics.inc'
530  include 'basicsp.inc'
531 C
532 C Two common blocks below are used in several subsequent routines.
533  parameter(nxx2m=nxxm*nxxm)
534  parameter(nxx22=2*nxx2m)
535  COMMON /ctmp1x/ crs(4,nxx2m),r1(nxxm),s1(nxxm),rxvr(nxxm,nxxm)
536  $ ,twrk(nxx2m),xyz(4,nxx22),break0,break1
537  $ , wkk(nxx2m),intrp(nxxm,nxm,2)
538  COMMON /ctmp2/ ncrs,nctr,ictr(50),nxvr(nxxm),indx(nxxm)
539  $ ,icelmn,icelmx,jcelmn,jcelmx
540  LOGICAL IFIN
541  dimension ivec(4),jvec(4),nrmout(2,4)
542  SAVE ivec,jvec,nrmout
543  DATA ivec / 0 , 1 , 1 , 0 /
544  DATA jvec / 0 , 0 , 1 , 1 /
545  DATA nrmout / 0,-1,1,0,0,1,-1,0/
546 C
547 C
548  ifedgb=.false.
549  newedg=1
550 C
551  icrn=iedgc
552  icrn=mod1(icrn,4)
553  i0 = icel + ivec(icrn)
554  j0 = jcel + jvec(icrn) + jcel0
555 C
556 C===================================================
557 C MAIN FOUR-EDGE LOOP
558 C===================================================
559 C
560  DO 1000 icornr=1,4
561 C
562 C Now we check the 4 corners of this cell
563 C until we find one that is IN.
564 C
565  icrn=iedgc+icornr
566  icrn=mod1(icrn,4)
567  i = icel + ivec(icrn)
568  j = jcel + jvec(icrn) + jcel0
569  IF (ifcrn.AND.icornr.EQ.1.AND..NOT.ifin(wrk(i,j))) THEN
570 C
571 C We've got a partial edge.
572 C
573  IF (iedgc.EQ.1.OR.iedgc.EQ.3) THEN
574 c write(77,*) 'call 3'
575  CALL fndrr(rc,r1(i0),r1(i),wrk(i0,j),wrk(i,j),val)
576  sc=s1(j)
577  ELSE
578  rc=r1(i)
579 c write(77,*) 'call 4'
580  CALL fndrr(sc,s1(j0),s1(j),wrk(i,j0),wrk(i,j),val)
581  ENDIF
582  ifedgb=.true.
583  ifcrn =.false.
584  return
585  ENDIF
586 C
587  IF (ifin(wrk(i,j))) THEN
588  IF (icornr.EQ.1) THEN
589 C
590 C We're on an edge, advance to next cell.
591 C
592  rc=r1(i)
593  sc=s1(j)
594  nxtedg=icrn
595  icel=icel+nrmout(1,nxtedg)
596  jcel=jcel+nrmout(2,nxtedg)
597 C Check cell advance past boundaries.
598  IF (icel.LT.icelmn.OR.icel.GT.icelmx.OR.
599  $ jcel.LT.jcelmn.OR.jcel.GT.jcelmx ) THEN
600  iedgc=nxtedg
601  icel = max(icel,icelmn)
602  icel = min(icel,icelmx)
603  jcel = max(jcel,jcelmn)
604  jcel = min(jcel,jcelmx)
605  ENDIF
606  ifedgb=.true.
607  ifcrn =.true.
608 C
609  ELSE
610 C Standard case:
611 C
612 C Find cross point
613 C
614  IF (newedg.EQ.1.OR.newedg.EQ.3) THEN
615 c write(77,*) 'call 5'
616  CALL fndrr(rc,r1(i0),r1(i),wrk(i0,j),wrk(i,j),val)
617  sc=s1(j)
