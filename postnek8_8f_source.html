2016  function gs_smooth(x,y,z,b,c,a,ja,ia,n,g,gg,s,eps,icall)

2019 c
2020 c
2021 c Input parameters
2022 c
2023 c
2024 c b(i) -- boundary (and box) information, b(i)>0 ==> point doesn't move
2025 c icall -- 1 ==> compute initial distance, a(ij)
2026 c s -- stretch. s=.85 implies that a(ij)/d_final = 1./0.85.
2027 c eps -- scaling factor for amount of movement on this pass
2028 c g(i) -- graph distance from node i to boundary

2036 c

2040  do ij=j0,j1
2041  j = ja(ij)
2042  if (j.ne.i) then
2043  dx = (x(i)-x(j))
2044  dy = (y(i)-y(j))
2045  dz = (z(i)-z(j))
2046  dx2 = dx*dx + dy*dy + dz*dz
2047  ds = sqrt(dx2)

2050 c Increase the stiffness for links connected to the boundary?
2051 c if (b(i).eq.1.or.b(j).eq.1) c(ij) = 2.

2054  else

2059  endif
