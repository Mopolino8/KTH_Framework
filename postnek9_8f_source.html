571  subroutine genstrv
572 C
573 C Generate Stress Vectors on selected planes
574 C
575  include 'basics.inc'
576  include 'basicsp.inc'
577  CHARACTER KEY,STRING*5,ELE(4),CVEL*20
578  parameter(nxm2=nxm*nym)
579  parameter(nxm3=nxm*nym*nzm)
580  COMMON /scrtch/ xplot(nxm2),yplot(nxm2),zplot(nxm2)
581  $ ,xvec(nxm2) ,yvec(nxm2) ,zvec(nxm2)
582  $ ,dudr(nxm3) ,duds(nxm3) ,dudt(nxm3)
583  $ ,dxds(nxm2) ,dyds(nxm2) ,dzds(nxm2)
584  dimension dxdr(nxm2) ,dydr(nxm2) ,dzdr(nxm2)
585  $ ,drdn(nxm2) ,dsdn(nxm2) ,dtdn(nxm2)
586  equivalence(dxdr,dudr),(dydr,duds),(dzdr,dudt)
587  equivalence(dxds,drdn),(dyds,dsdn),(dzds,dtdn)
588 c
589  LOGICAL IFCRV(4),IFNX
590  ind(i,j,k,iel)=i+nx*(j-1)+nx*ny*(k-1)+nx*ny*nz*(iel-1)
591 C
592  nxyz = nx*ny*nz
593  uvmax=0.0
594  IF (if3d) THEN
595 C
596 C Sort the selected list of planes according to visibility
597  CALL sortl
598 C
599 C Plot the selected planes
600  DO 9000 ip=1,nlstp
601 C
602  lista=abs(list(ip))
603  CALL decod(iplane,ipln,iel,idum,lista,nx,6,nelm)
604 c write(6,*) 'n',nlstp,iel,ipln,iplane
605 c
606  i1=1
607  i2=nx
608  j1=1
609  j2=ny
610  k1=1
611  k2=nz
612 c
613 C Plot them all
614  i3=1
615  j3=1
616  k3=1
617  nxskip=nx
618 c
619  IF (ipln.LE.2) THEN
620  i1=iplane
621  i2=iplane
622  i3=1
623  ELSEIF (ipln.LE.4) THEN
624  j1=iplane
625  j2=iplane
626  j3=1
627  ELSE
628  k1=iplane
629  k2=iplane
630  k3=1
631  ENDIF
632  ii=0
633  DO 100 k=k1,k2,k3
634  DO 100 j=j1,j2,j3
635  DO 100 i=i1,i2,i3
636  ipoint=ind(i,j,k,iel)
637  ii=ii+1
638  xplot(ii)=xp(ipoint)
639  yplot(ii)=yp(ipoint)
640  zplot(ii)=zp(ipoint)
641  100 CONTINUE
642  nxy = ii
643 C
644 C Compute normal vector
645 C
646  call mxm(dgdr,nx,xplot ,nx,dxdr,nx)
647  call mxm(dgdr,nx,yplot ,nx,dydr,nx)
648  call mxm(dgdr,nx,zplot ,nx,dzdr,nx)
649  call mxm(xplot,nx,dgdrt,nx,dxds,nx)
650  call mxm(yplot,nx,dgdrt,nx,dyds,nx)
651  call mxm(zplot,nx,dgdrt,nx,dzds,nx)
652  call vcross
653  $ (xvec,yvec,zvec,dxdr,dydr,dzdr,dxds,dyds,dzds,nxy)
654  alpha=1.0
655  if(ipln.eq.2.or.ipln.eq.3.or.ipln.eq.6) then
656 c left-handed
657  alpha = -alpha
658  endif
659  call vnorms(xvec,yvec,zvec,alpha,nxy)
660 c
661  IF(quanty.EQ.'NORMAL VEC') THEN
662  ii = 0
663  call dudrst(dudr,duds,dudt,u(ieoff))
664  DO 190 k=k1,k2,k3
665  DO 190 j=j1,j2,j3
666  DO 190 i=i1,i2,i3
667  ipoint=ind(i,j,k,iel)
668  ii=ii+1
669  wkv1(ipoint) = xvec(ii)
670  wkv2(ipoint) = yvec(ii)
671  wkv3(ipoint) = zvec(ii)
672  190 CONTINUE
673  ELSE
674 c
675 c Normals are computed... now compute du/dn along normal
676 c NOTE: We rely upon incompressibility to annihilate the
677 c derivative of the normal velocity wrt the normal
678 c direction.
679 c
680 c Compute du/dn near surface with spectral formulation
681 c
682  ieoff=nxyz*(iel-1)+1
683 c
684  ii = 0
685  DO 200 k=k1,k2,k3
686  DO 200 j=j1,j2,j3
687  DO 200 i=i1,i2,i3
688  ii=ii+1
689  ipoint=ind(i,j,k,iel)
690  drdn(ii) = rxm1(ipoint)*xvec(ii)
691  $ + rym1(ipoint)*yvec(ii)
692  $ + rzm1(ipoint)*zvec(ii)
693  dsdn(ii) = sxm1(ipoint)*xvec(ii)
694  $ + sym1(ipoint)*yvec(ii)
695  $ + szm1(ipoint)*zvec(ii)
696  dtdn(ii) = txm1(ipoint)*xvec(ii)
697  $ + tym1(ipoint)*yvec(ii)
