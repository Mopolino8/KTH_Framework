706  subroutine gaujordf(a,m,n,indr,indc,ipiv,ierr)
707 C
708 C Gauss-Jordan matrix inversion with full pivoting
709 c
710 c Num. Rec. p. 30, 2nd Ed., Fortran
711 c
712 C
713 C A is an nxn matrix, assumed contiguous in memory
714 C W is a work array of dimension n
715 C
716  real a(m,n)
717  integer indr(m),indc(n),ipiv(n)
718 c
719  real rmult
720  integer wdsize
721 c
722  ierr = 0
723 c
724  eps = 1.e-12
725  one = 1.
726  wdsize = 8
727  if (one+eps.eq.1.) wdsize = 4
728 c
729  call izero(ipiv,n)
730 c
731 c
732  do j=1,n
733  amx=0.
734 c
735 c Pivot search
736  do i=1,m
737  if (ipiv(i).ne.1) then
738  do k=1,n
739  if (ipiv(k).eq.0) then
740  if (abs(a(i,k)).ge.amx) then
741  amx = abs(a(i,k))
742  ir = i
743  jc = k
744  endif
745  elseif (ipiv(k).gt.1) then
746  ierr = -ipiv(k)
747  return
748  endif
749  enddo
750  endif
751  enddo
752  ipiv(jc) = ipiv(jc) + 1
753 c
754 c Swap rows
755  if (ir.ne.jc) then
756  do k=1,n
757  tmp = a(ir,k)
758  a(ir,k) = a(jc,k)
759  a(jc,k) = tmp
760  enddo
761  endif
762  indr(j)=ir
763  indc(j)=jc
764  if (abs(a(jc,jc)).lt.eps) then
765  write(6,*) 'small Gauss Jordan Piv:',jc,a(jc,jc)
766  ierr = jc
767  return
768  endif
769  piv = 1./a(jc,jc)
770  a(jc,jc)=1.
771  do k=1,n
772  a(jc,k) = a(jc,k)*piv
773  enddo
774 c
775  do i=1,m
776  if (i.ne.jc) then
777  rmult = a(i,jc)
778  a(i,jc) = 0.
779  do k=1,n
780  a(i,k) = a(i,k) - rmult*a(jc,k)
781  enddo
782  endif
783  enddo
784  enddo
785 c
786 c Unscramble matrix
787  do k=n,1,-1
788  if (indr(k).ne.indc(k)) then
789  do i=1,m
790  tmp=a(i,indr(k))
791  a(i,indr(k))=a(i,indc(k))
792  a(i,indc(k))=tmp
793  enddo
794  endif
795  enddo
796 c
797  return
798  end
