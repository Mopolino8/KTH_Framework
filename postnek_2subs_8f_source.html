264  SUBROUTINE roots(XVAL,NROOTS,FTARGT,F,N)
265 C
266 C Find all values of I which yield F(I) close to FTARGT
267 C and give the additional increment DI such that F(I+DI)+FTARGT,
268 C where F(I+DI) is interpreted as the linear interpolation of F.
269 C
270  dimension xval(2,n),f(0:n)
271 C
272  nroots=0
273  DO 100 i=1,n
274  f1=f(i-1)
275  f2=f(i)
276  df1=ftargt-f1
277  df2=f2-ftargt
278  IF (df1*df2.GE.0.0) THEN
279  nroots=nroots+1
280  df=f2-f1
281  IF (df.EQ.0) THEN
282  di=0.5
283  ELSE
284  di=(ftargt-f1)/df
285  ENDIF
286  xval(1,nroots)=float(i-1)
287  xval(2,nroots)=di
288  ENDIF
289  100 CONTINUE
290  RETURN
291  END

309  SUBROUTINE convert(T)
310 C
311  CHARACTER*12 VC
312  CHARACTER*1 V1(12)
313  equivalence(v1,vc)
314  CHARACTER*1 CSIGN
315 C
316  REAL*4 W
317  CHARACTER*4 WC
318  CHARACTER*1 W1(4)
319  equivalence(w1,w)
320  equivalence(w1,wc)
321 C
322  CHARACTER*1 ALPH64(0:63)
323  SAVE alph64
324  DATA alph64
325  $ /'1','2','3','4','5','6','7','8','9','0'
326  $ ,'a','b','c','d','e','f','g','h','i','j'
327  $ ,'k','l','m','n','o','p','q','r','s','t'
328  $ ,'u','v','w','x','y','z'
329  $ ,'A','B','C','D','E','F','G','H','I','J'
330  $ ,'K','L','M','N','O','P','Q','R','S','T'
331  $ ,'U','V','W','X','Y','Z','+','-'/
332 C
333 C Find out the usual decimal format for T
334 C
335  WRITE(vc,10) t
336  10 FORMAT(e12.5)
337 C
338 C Begin converting the mantissa to base 64
339 C
340  READ(vc,11) mantis
341  11 FORMAT(3x,i5)
342 C
343 C Sign?
344 C
345  READ(vc,12) csign
346  12 FORMAT(a1)
347  IF (csign.EQ.'-') mantis=-mantis
348  mantis=mantis+131072
349 C
350 C ONES,TENS, HUNDREDS
351 C
352  ione=mod(mantis,64)
353  itmp=mantis/64
354  iten=mod(itmp,64)
355  itmp=itmp/64
356  ihun=mod(itmp,64)
357 C
358 C Exponent
359 C
360  READ(vc,21) iexp
361  21 FORMAT(9x,i3)
362 C We assume that the exponent is bounded by 31.
363  iexp=iexp+31
364 C
365 C Compute alpha equivalent
366 C
367  w1(1)=alph64(ihun)
368  w1(2)=alph64(iten)
369  w1(3)=alph64(ione)
370  w1(4)=alph64(iexp)
371 C
372 C Convert the input value
373  t=w
374  RETURN
375  END
