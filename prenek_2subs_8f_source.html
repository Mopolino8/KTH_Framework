332  subroutine roots(xval,nroots,ftargt,f,n)
333 C
334 C Find all values of I which yield F(I) close to FTARGT
335 C and give the additional increment DI such that F(I+DI)+FTARGT,
336 C where F(I+DI) is interpreted as the linear interpolation of F.
337 C

339 C

342  f1=f(i-1)
343  f2=f(i)
344  df1=ftargt-f1
345  df2=f2-ftargt
346  IF (df1*df2.GE.0.0) THEN
347  nroots=nroots+1

350  di=0.5
351  ELSE

353  endif
