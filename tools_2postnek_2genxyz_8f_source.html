593  subroutine prjects(x0,dxc,c,cc)
594 c
595 c Project the point x0 onto surface described by characteristics
596 c given in the array c and cc.
597 c
598 c dxc - characteristic length scale used to estimate gradient.
599 c
600  real x0(3)
601  real c(5)
602  character*1 cc
603  real x1(3)
604  logical if3d
605 c
606  if3d = .true.
607  if (dxc.le.0.0) then
608  write(6,*) 'invalid dxc',dxc,x0
609  write(6,*) 'Abandoning prjects'
610  return
611  endif
612 c
613  call copy(x1,x0,3)
614  r0 = ressrf(x0,c,cc)
615  if (r0.eq.0) return
616 c
617 c Must at least use ctr differencing to capture symmetry!
618 c
619  x1(1) = x0(1) - dxc

621  x1(1) = x0(1) + dxc
622  r2 = ressrf(x1,c,cc)
623  x1(1) = x0(1)
624  rx = 0.5*(r2-r1)/dxc
625 c

628  x1(2) = x0(2) + dxc
629  r2 = ressrf(x1,c,cc)/dxc
630  x1(2) = x0(2)
631  ry = 0.5*(r2-r1)/dxc
632 c
633  if (if3d) then
634  x1(3) = x0(3) - dxc
635  r1 = ressrf(x1,c,cc)/dxc
636  x1(3) = x0(3) + dxc
637  r2 = ressrf(x1,c,cc)/dxc
638  rz = 0.5*(r2-r1)/dxc
639  endif

641  alpha = - r0/rnorm2
642 c
643 c Apply secant method: Use an initial segment twice expected length
644 c
645  x1(1) = x0(1) + 2.0*rx * alpha
646  x1(2) = x0(2) + 2.0*ry * alpha
647  x1(3) = x0(3) + 2.0*rz * alpha
648  call srfind(x1,x0,c,cc)

650 c write(38,38) x1,x0,cc,c(2),c(3)
651 c 38 format(3f9.6,1x,3f9.6,1x,a1,2f6.2)
652 c
653  call copy(x0,x1,3)
654  return
655  end

657  subroutine srfind(x1,x0,c,cc)
658  real x1(3),x0(3)
659  real c(5)

661 c
662 c Find point on line segment that intersects the ellipsoid
663 c specified by:
664 c (x/a)**2 + (y/b)**2 + (z/b)**2 = 1
665 c
666 c
667 c Algorithm: 4 rounds of secant x_k+1 = x_k - f/f'
668 c

677  do 10 i=1,9

682  a0 = a1

684  endif

686  x1(2) = x0(2) + a1*dy
687  x1(3) = x0(3) + a1*dz
688  10 continue

690  return
691  end
