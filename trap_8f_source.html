1125  subroutine fd_weights_full(xx,x,n,m,c)
1126 c
1127 c This routine evaluates the derivative based on all points
1128 c in the stencils. It is more memory efficient than "fd_weights"
1129 c
1130 c This set of routines comes from the appendix of
1131 c A Practical Guide to Pseudospectral Methods, B. Fornberg
1132 c Cambridge Univ. Press, 1996. (pff)
1133 c
1134 c Input parameters:
1135 c xx -- point at wich the approximations are to be accurate
1136 c x -- array of x-ordinates: x(0:n)
1137 c n -- polynomial degree of interpolant (# of points := n+1)
1138 c m -- highest order of derivative to be approxxmated at xi
1139 c
1140 c Output:
1141 c c -- set of coefficients c(0:n,0:m).
1142 c c(j,k) is to be applied at x(j) when
1143 c the kth derivative is approxxmated by a
1144 c stencil extending over x(0),x(1),...x(n).
1145 c
1146 c
1147  real x(0:n),c(0:n,0:m)
1148 c
1149  c1 = 1.
1150  c4 = x(0) - xx
1151 c
1152  do k=0,m
1153  do j=0,n
1154  c(j,k) = 0.
1155  enddo
1156  enddo
1157  c(0,0) = 1.
1158 c
1159  do i=1,n
1160  mn = min(i,m)
1161  c2 = 1.
1162  c5 = c4
1163  c4 = x(i)-xx
1164  do j=0,i-1
1165  c3 = x(i)-x(j)
1166  c2 = c2*c3
1167  do k=mn,1,-1
1168  c(i,k) = c1*(k*c(i-1,k-1)-c5*c(i-1,k))/c2
1169  enddo
1170  c(i,0) = -c1*c5*c(i-1,0)/c2
1171  do k=mn,1,-1
1172  c(j,k) = (c4*c(j,k)-k*c(j,k-1))/c3
1173  enddo
1174  c(j,0) = c4*c(j,0)/c3
1175  enddo
1176  c1 = c2
1177  enddo
1178  s0 = 0.
1179  do i=0,n
1180  s0 = s0+c(i,0)
1181  enddo
1182 c write(6,2) 'fd wt sum:',s0
1183  1 format('c',10f8.4)
1184  2 format(a10,10f8.4)
1185  return
1186  end
